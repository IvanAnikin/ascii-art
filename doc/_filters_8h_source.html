312class ConvolutionFilter : public Filter {
313public:
314 ConvolutionFilter(bool active) : Filter(active) {}
315
316 void apply(unsigned char* m_pixels, int width, int height) override {
317 int kernelSize = 3;
318 double kernel[3][3] = {{-1, -1, -1},
319 {-1, 8, -1},
320 {-1, -1, -1}};
321
322 // create a temporary copy of the image
323 unsigned char* tempPixels = new unsigned char[width * height * 3];
324 std::copy(m_pixels, m_pixels + width * height * 3, tempPixels);
325
326 // loop through all pixels in the image
327 for (int y = 0; y < height; y++) {
328 for (int x = 0; x < width; x++) {
329 // apply the convolution kernel to the pixel
330 double sumR = 0, sumG = 0, sumB = 0;
331 for (int j = 0; j < kernelSize; j++) {
332 for (int i = 0; i < kernelSize; i++) {
333 int pixelX = std::min(std::max(x + i - kernelSize / 2, 0), width - 1);
334 int pixelY = std::min(std::max(y + j - kernelSize / 2, 0), height - 1);
335 int pixelIndex = (pixelY * width + pixelX) * 3;
336 double kernelValue = kernel[j][i];
337 sumR += tempPixels[pixelIndex] * kernelValue;
338 sumG += tempPixels[pixelIndex + 1] * kernelValue;
339 sumB += tempPixels[pixelIndex + 2] * kernelValue;
340 }
341 }
342
343 // write the convolved pixel back to the image
344 int pixelIndex = (y * width + x) * 3;
345 m_pixels[pixelIndex] = std::max(0.0, std::min(255.0, sumR));
346 m_pixels[pixelIndex + 1] = std::max(0.0, std::min(255.0, sumG));
347 m_pixels[pixelIndex + 2] = std::max(0.0, std::min(255.0, sumB));
348 }
349 }
350
351 // free the temporary copy of the image
352 delete[] tempPixels;
353 }
354 void printStartCode(std::ostream& os) const override {}
355 void printEndCode(std::ostream& os) const override {}
356
357 void update(const std::string& command) override { }
358 double get_value() override { return 0.0; }
359
360 std::ostream& print(std::ostream& os) const override {
361 os << "Convolution\t\t-\tModify using:\t'convolution'";// << (isActive ? "on" : "off") << std::endl;
362 return os;
363 }
364 void save(std::ofstream& stream) const override {
365 stream << "ConvolutionFilter " << std::boolalpha << isActive << std::endl;
